**콜백** - 인라인 방식으로 다시 작성하기
{% highlight javascript %}
var fs = require("fs");
function readingfile(callback){
	fs.readFile("readme.txt","utf-8",function(err,content){
		callback(content);
	})
};
readingfile(function(fcontent){
	console.log(fcontent);
});
{% endhighlight %}
